2) Security + performance headers (Edge)

Edit: next.config.js (add headers)

/** @type {import('next').NextConfig} */
const nextConfig = {
  experimental: { optimizePackageImports: ["lucide-react"] },
  images: { formats: ["image/avif","image/webp"] },
  async headers() {
    return [
      {
        source: "/:path*",
        headers: [
          { key: "Strict-Transport-Security", value: "max-age=63072000; includeSubDomains; preload" },
          { key: "X-Content-Type-Options", value: "nosniff" },
          { key: "X-Frame-Options", value: "SAMEORIGIN" },
          { key: "Referrer-Policy", value: "strict-origin-when-cross-origin" },
          { key: "Permissions-Policy", value: "camera=(), microphone=(), geolocation=()" }
        ]
      }
    ];
  }
};
module.exports = nextConfig;

7) Workflow dashboard first-paint and empty state

Edit: src/app/workflow/page.tsx (add skeleton + empty)

import { Suspense } from "react";

async function Grid() {
  const res = await fetch(`${process.env.NEXT_PUBLIC_BASE_URL ?? ""}/api/workflows`, { cache: "no-store" });
  if (res.status === 401) return null;
  const { items = [] } = await res.json().catch(()=>({items:[]}));
  if (!items.length) {
    return <div className="rounded-2xl border p-6 text-sm text-neutral-600">No workflows yet. Create one from the builder.</div>;
  }
  return (
    <div className="grid sm:grid-cols-2 lg:grid-cols-3 gap-4">
      {items.map((w:any)=>(
        <div key={w.id} className="rounded-2xl border p-4">
          <div className="text-base font-semibold">{w.name ?? "Workflow"}</div>
          <div className="text-xs mt-1 opacity-70">{w.status ?? "idle"}</div>
        </div>
      ))}
    </div>
  );
}

export default function Page() {
  return (
    <main className="max-w-6xl mx-auto px-4 py-8">
      <h1 className="text-2xl font-semibold mb-4">Workflows</h1>
      <Suspense fallback={<div className="h-24 rounded-2xl border animate-pulse" />}>
        {/* @ts-expect-error Async Server Component */}
        <Grid />
      </Suspense>
    </main>
  );
}

Deploy.
Mobile check: after sign-in, /workflow shows skeleton then grid or clean empty state.
