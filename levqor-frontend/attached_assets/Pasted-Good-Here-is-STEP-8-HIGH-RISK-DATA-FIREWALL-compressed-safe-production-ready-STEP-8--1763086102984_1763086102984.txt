STEP 8 — HIGH-RISK DATA FIREWALL (COMPRESSED VERSION)

1. Backend Rule — Reject Prohibited Data

Add to all workflow creation & task execution endpoints:

// forbidden terms list
const BLOCKED = [
  "medical", "diagnosis", "treatment", "symptom",
  "legal advice", "lawsuit", "attorney", "contract drafting",
  "tax advice", "financial advice", "investment", "trading signals",
  "credit score", "loan approval"
];

function containsBlocked(text: string) {
  return BLOCKED.some(x => text.toLowerCase().includes(x));
}

// middleware
if (containsBlocked(req.body.description) || containsBlocked(JSON.stringify(req.body))) {
  return res.status(400).json({
    ok: false,
    error: "This type of workflow is not permitted. High-risk categories are blocked for safety."
  });
}
